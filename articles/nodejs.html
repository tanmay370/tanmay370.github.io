Server Side code Snipped

30 August 2023 by Tanmay

                // Import Module, IP, Port
                const http = require ('http');
                const hostname = '127.0.0.1';
                const port = 3000;

                // Create a server
                const server = http.createServer((req, res) => {
                    res.statusCode = 200;
                    res.setHeader('Content-Type', 'text/plain');
                    res.end('Hello World');
                });
                // Listening to server
                server.listen(port, hostname, () => {
                    console.log(`Server running at http://${hostname}:${port}/`);
                });

Routing [Routes/Pages]

                // Create a server
                const server = http.createServer((req, res) => {
                    res.statusCode = 200;
                    res.setHeader('Content-Type', 'text/plain');
                    res.end('Hello World');
                });
                // Create routes based on URL
                    if (url == '/'){
                        res.end(home);
                    }
                    else if (url == '/about'){
                        res.end(about);
                    }
                    else {
                        res.StatusCode = 404;
                        res.end(404 Page Not Found );
                    }
                // Listening to server
                server.listen(port, hostname, () => {
                    console.log(`Server running at http://${hostname}:${port}/`);
                });

Express

31 August 2023 by Tanmay

                // Create a server
                const express = require('express');
                const app = express();
                const port = 3000;
                // Create routes based on URL
                app.get('/', (req, res) => {
                    res.send('Hello World!');
                });
                // Listening to server
                app.listen(port, () => {
                    console.log(`Example app listening on port ${port}`);
                });

Node Package Manager [npm]

npm is NodeJs package manager for maintaing modules and dependencies

            node --version -->  Check version of node installed in your computer
            npm --version  -->  Check version of npm installed in your computer
            npm init       -->  To install, and make package.json file
            npm install    -->  Used when to install module and their dependencies exists in package.json file
